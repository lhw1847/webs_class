while문을 이용해서 구구단 출력

{
    let num1 = 2;
    let num2 = 1;

    while (num1 <= 9) {
        num2 = 1;
        while (num2 <= 9) {
            document.write(num1 + " * " + num2 + " = " + (num1 * num2) + "<br>");
            num2++;
        }
        num1++;
        document.write("<br>");
    }
}
